#version 450

layout(location = 0)
	out vec4 cFragColor;
	
in vec4 variPosition;	
in vec2 vTexCoord;
in vec2 vAltTexCoord;
uniform bvec4 keysWASD;
uniform sampler2D uTexture;
layout(binding = 1)
	uniform sampler2D uTextureTwo;
vec4 changeColWASD(vec4 defaultCol)
{
	//W
	if(keysWASD.x)
	{
		return vec4(1.0,0.5,0.0,1.0);
	}
	//A
	if(keysWASD.y)
	{
		return vec4(0.0,0.0,0.0,1.0);
	}
	//S
	if(keysWASD.z)
	{
		return texture(uTextureTwo, vAltTexCoord);
	}
	//D
	if(keysWASD.w)
	{
		return texture(uTexture, vTexCoord);
	}
	return defaultCol;

}

void main()
{
	cFragColor = changeColWASD(vec4(0.5));
}
